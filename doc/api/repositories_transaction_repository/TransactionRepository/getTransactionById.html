Future<Transaction?> getTransactionById(int id) async {
  final db = await dbHelper.db;
  final List<Map<String, dynamic>> maps = await db.query(
    'transactions',
    where: 'id = ?',
    whereArgs: [id],
  );
  if (maps.isNotEmpty) return Transaction.fromMap(maps.first);
  return null;
}
